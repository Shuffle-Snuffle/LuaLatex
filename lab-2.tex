	label=listing:dichotomy,
        captionpos=tr,
	]
dichotomy <- function(f, a, b, epsilon) {
    repeat {
        x <- (a + b) / 2
        y <- f(x)
        ifelse (f(a) * y < 0, b <- x, a <- x)
        if (abs(y) <= epsilon) {
            break
        }
    }
    x
}
\end{lstlisting}
